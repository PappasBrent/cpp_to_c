start               =   translation_unit;

translation_unit    =   declaration_list;

declaration_list    =   declaration
                    |   declaration, declaration_list;

declaration         =   'int', identifier, ';'
                    |   'int', identifier, '=', expr, ';';

identifier          =   ? A valid C identifier ?;

expr                =   constant
                    |   identifier;

constant            =   ? any C integer constant ?;
